#Election#
country_event = {
	type = country_event
	id = srd.1
	title = srd.election.t
	desc = srd.election.d
	picture = GFX_report_event_election_vote

	trigger = {
		tag = SRD
		date > 1954.1.1
	}

	fire_only_once = yes

	mean_time_to_happen = {
		hour = 1
	}
	
	option = {
		name = srd.election.a
			custom_effect_tooltip = SRD_election_tt
		hidden_effect = {
			complete_national_focus = SRD_elections_focus
			promote_character = SRD_Anton
			retire_character = SRD_Alphy
			SRD = { country_event = { id = srd.3 days = 25 } }
		}
	}
	option = {
		name = srd.election.b
			custom_effect_tooltip = SRD_election2_tt
			add_timed_idea = { idea = SRD_confusion days = 150 }
		hidden_effect = {
			set_politics = { 
			ruling_party = liberal
			elections_allowed = yes
			}
				add_popularity = {  
				ideology = liberal
				popularity = 0.1
			}
			complete_national_focus = SRD_elections_focus
			promote_character = SRD_Anton
			retire_character = SRD_Alphy
			SRD = { country_event = { id = srd.3 days = 25 } }
		}
	}
}
